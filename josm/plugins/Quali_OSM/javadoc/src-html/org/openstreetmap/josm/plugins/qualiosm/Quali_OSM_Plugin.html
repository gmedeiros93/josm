001package org.openstreetmap.josm.plugins.qualiosm;

003import org.openstreetmap.josm.gui.MainApplication;
004import org.openstreetmap.josm.plugins.Plugin;
005import org.openstreetmap.josm.plugins.PluginInformation;
006import org.openstreetmap.josm.gui.help.HelpUtil;
007import static org.openstreetmap.josm.tools.I18n.tr;

012public class Quali_OSM_Plugin extends Plugin {

025

034     
035      
036    }

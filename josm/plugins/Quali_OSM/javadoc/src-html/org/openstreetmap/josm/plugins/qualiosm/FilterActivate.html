013import org.openstreetmap.josm.gui.layer.ImageryLayer;
014import org.openstreetmap.josm.gui.layer.Layer;

018/**
019 * This filters is responsible for creating filter's dialog where user can
020 * choose and add new filter at this dialog.
021 *
022 * @author Gabriel Medeiros
023 */
024public class FilterActivate {
025
026  
027    
028    private Layer layer;
029    private FiltersManager filtersManager;
030   
031   
032
033    public FilterActivate(ImageryLayer layer) {
034        
035        this.layer = layer;
036        this.filtersManager = new FiltersManager(this);
037        layer.addImageProcessor(filtersManager);
038    }
039
040    FilterActivate() {
041        throw new UnsupportedOperationException("Not supported yet."); 
042    }
043
044}
